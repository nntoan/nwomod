province_event = { # Demand Autonomy
  id = 15500
  title = "EVT_15500_NAME"
  desc = "EVT_15500_DESC"
  picture = "demand_autonomy"

  trigger = {
    is_primary_culture = no
    is_accepted_culture = no
    controlled_by = owner
    owner = {
      OR = {
        has_country_flag = sectarianism
        immigration_policy = naturalization
        colonial_nation = yes
      }
    }
    NOT = { is_core = THIS }
    NOT = { has_province_modifier = grant_independence }
    NOT = { has_province_modifier = promised_autonomy }
    NOT = { has_province_modifier = denied_independence }
  }

  mean_time_to_happen = {
    months = 960
    modifier = {
      factor = 0.5
      any_neighbor_province = {
        has_province_modifier = promised_autonomy
      }
    }
    modifier = {
      factor = 0.8
      state_scope = { flashpoint_tension  = 10 }
    }
    # a dominating majority primary culture makes us even more likely to request autonomy
    modifier = {
      factor = 0.5
      is_primary_culture = no
      is_accepted_culture = no
    }
    modifier = {
      factor = 0.9
      average_militancy = 5
    }
    modifier = {
      factor = 0.9
      average_consciousness = 5
    }
    modifier = {
      factor = 1.5
      NOT = {
        AND = {
          average_consciousness = 5
          average_militancy = 5
        }
      }
    }
    modifier = {
      factor = 1.5
      owner = { NOT = { neoprotectionism = 1 } }
    }
    modifier = {
      factor = 4
      owner = { NOT = { immigration_policy = naturalization } }
    }
  }

  option = {
    name = "EVT_15500_B"		# Promised autonomy
    any_pop = {
      militancy = -2
      consciousness = 0.5
    }
    add_province_modifier = { name = promised_autonomy duration = 1080 }
    ai_chance = { factor = 0.5 modifier = { NOT = { immigration_policy = naturalization } factor = 0  } }
  }
  option = {
    name = "EVT_15500_A"		#Deny Autonomy
    any_pop = {
      militancy = 1
      consciousness = 1
    }
    add_province_modifier = { name = denied_independence duration = 1080 }
    ai_chance = { factor = 0.5 modifier = { immigration_policy = naturalization factor = 0 } }
  }
}
